#include<algorithm>
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        if(target<=nums[0])return 0;
        if(target>nums[(nums.end()-nums.begin()-1)]) return nums.size();
        if(target==nums[(nums.end()-nums.begin()-1)]) return nums.size()-1;
        return (lower_bound(nums.begin(),nums.end(),target)-nums.begin());
       
    }
};
