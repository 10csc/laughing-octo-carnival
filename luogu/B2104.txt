#include<iostream>
using namespace std;
int nums[101][101];
int m, n;
int main() 
{
	ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);
	cin >> m >> n;
	for (int i = 0; i < m; i++)
	{
		for (int j = 0; j < n; j++)
		{
			cin >> nums[i][j];
		}
	}
	for (int i = 0; i < m; i++)
	{
		for (int j = 0; j < n; j++)
		{
			int a; cin >> a;
			nums[i][j] += a;
		}
	}
	for (int i = 0; i < m; i++)
	{
		for (int j = 0; j < n; j++)
		{
			cout << nums[i][j]<<' ';
		}
		cout << endl;
	}
	return 0;
}
