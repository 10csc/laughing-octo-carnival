#include<iostream>
using namespace std;
string game1[102];  //玩家1
string game2[102];
int n;
int ans(string a, string b)
{
	if (a == "Rock") { if (b == "Rock")return 0; if (b == "Paper")return -1; if (b == "Scissors") return 1; }
	else if (a == "Scissors") { if (b == "Rock")return -1; if (b == "Paper")return 1; if (b == "Scissors") return 0; }
	else { if (b == "Rock")return 1; if (b == "Paper")return 0; if (b == "Scissors") return -1; }
}
int main()
{
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> game1[i] >> game2[i];
	}
	for (int i = 0; i < n; i++)
	{
		int m=ans(game1[i], game2[i]);
		if (m == 0)cout << "Tie" << endl;
		else if (m == 1)cout << "Player1" << endl;
		else cout << "Player2" << endl;
	}
	return 0;
}
