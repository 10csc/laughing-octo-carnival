//问题的形式转化
#include<iostream>
using namespace std;
int main()
{
	int i = 1,ans = 0;
	int n; cin >> n;
	while (1)
	{
		ans += i;
		for (int j = 1; j <= i; j++)  //判断当前位置是否是家
		{
			if (3 * j == ans - n)
			{
				//是
				cout << j << ' ' << i;
				return 0;
			}
		}
		i++;
	}
}
